class Main {
    public static void main(String[] args) {

        for (char i = 'A'; i <= 'D'; i++) {

            for (int j = 0; j < 4; j++) {

                if ((i - 'A') % 2 == 0) {
                    System.out.print(Character.toLowerCase(i) + " ");
                } else {
                    System.out.print(i + " ");
                }

            }

            System.out.println();
        }
    }
}
